# For Designer.cs files:
1. In InitializeComponent after "components = " line, add:
   themeManager1 = new ZidUtilities.CommonCode.Win.Controls.ThemeManager(components);

2. After SuspendLayout(), add:
   //
   // themeManager1
   //
   themeManager1.ParentForm = this;

3. Before ResumeLayout, add Load event:
   Load += [DialogName]_Load;

4. At end of field declarations, add:
   private ZidUtilities.CommonCode.Win.Controls.ThemeManager themeManager1;

# For .cs file, add Load handler:
private void [DialogName]_Load(object sender, EventArgs e)
{
    themeManager1.ApplyTheme(Properties.Settings.Default.SelectedTheme);
}
